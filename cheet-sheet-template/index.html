# Beginner Friendly Library Management System in Python using CRUD 

book_list = {}

print(f"{"_"*20} \n Instructions \n{"_"*20}")


# adding book
while True:
req = input("Enter your choice (' list ' 'addone' 'addmany' 'check' 'delete' 'exit'): ").lower()
if req == "list":
    print(f"{"_"*20}\nBook List\n{"_"*20}")
    for j in book_list.keys():
        print(f"Book Name : {j} | {" "*(50-len(j))} Author : {book_list[j]["Author"]} ")

elif(req=="addone"):
    book_name = input(" Enter book name : ")
    book_author  = input(" Enter author name : ")
    no_of_book = int(input("Enter number of books : "))
    book_list[book_name ] = {"Author":book_author, "Count":no_of_book}
    print("Added Successfully, please type 'list' for you confirmation")

# Adding many book

elif(req=="addmany"):
    while True:
        book_name = input("Enter book name ( type 'exit' if you donot have any book left to add)")
        if(book_name=="exit"):
            break
        else:
            book_author = input(" Enter author name : ")
            no_of_book = int(input("Enter number of books : "))
            book_list[book_name ] = {"Author":book_author, "Count":no_of_book}

# Deleting item

elif(req=="delete"):
    name = input("Enter book name : ")
    book_list.pop(name)
    print("Deleted successfully please type 'list' for you confirmation")

# checking books

elif (req == "check"):
    name = input(" Enter the book name ")
    if name in book_list.keys():
        print(" Yes the book is there in the list")
        print(book_list[name])
    else:
        print("Book NOT Found")

# Going out of program
elif(req=="exit"):
    break
else:
    print("Invalid input\nTry again or type 'exit'")
